        // 获取当前场景键

        function getCurrentSceneKey() {

            return `year${gameState.year}-${months[gameState.month].toLowerCase()}`;

        }


        // 更新时间显示

        function updateTimeDisplay() {

            document.getElementById('month-year').textContent = 

                `${years[gameState.year]} | ${months[gameState.month]}`;

        }


        // 更新属性显示

        function updateStats() {

            document.getElementById('skill-value').textContent = gameState.englishSkill;

            document.getElementById('experience-value').textContent = gameState.experience;

            document.getElementById('network-value').textContent = gameState.network;

            document.getElementById('finance-value').textContent = gameState.finance;

            

            // 更新进度条

            document.getElementById('skill-progress').style.width = `${Math.min(100, Math.max(0, gameState.englishSkill))}%`;

            document.getElementById('experience-progress').style.width = `${Math.min(100, Math.max(0, gameState.experience))}%`;

            document.getElementById('network-progress').style.width = `${Math.min(100, Math.max(0, gameState.network))}%`;

            document.getElementById('finance-progress').style.width = `${Math.min(100, Math.max(0, gameState.finance / 20))}%`;

        }


        // 应用效果

        function applyEffect(effect) {

            if (effect.englishSkill) gameState.englishSkill += effect.englishSkill;

            if (effect.experience) gameState.experience += effect.experience;

            if (effect.network) gameState.network += effect.network;

            if (effect.finance) gameState.finance += effect.finance;

            

            // 确保属性不为负数

            gameState.englishSkill = Math.max(0, gameState.englishSkill);

            gameState.experience = Math.max(0, gameState.experience);

            gameState.network = Math.max(0, gameState.network);

            gameState.finance = Math.max(0, gameState.finance);

        }


        // 显示随机事件

        function showRandomEvent() {

            if (Math.random() > 0.7 && !gameState.eventOccurred) { // 30%概率触发事件

                const event = randomEvents[Math.floor(Math.random() * randomEvents.length)];

                document.getElementById('event-text').textContent = event.text;

                document.getElementById('event-display').style.display = 'block';

                applyEffect(event.effect);

                gameState.eventOccurred = true;

                updateStats();

                return true;

            }

            return false;

        }


        // 更新显示

        function updateDisplay() {

            updateTimeDisplay();

            updateStats();

            

            // 获取当前场景

            const sceneKey = getCurrentSceneKey();

            const scene = scenes[sceneKey] || scenes[gameState.currentScene];

            

            // 显示场景文本

            document.getElementById('story-text').textContent = scene.text;

            

            // 生成选择按钮

            const choicesContainer = document.getElementById('choices-container');

            choicesContainer.innerHTML = '';

            

            if (scene.choices) {

                scene.choices.forEach((choice, index) => {

                    // 检查条件（如果有的话）

                    if (!choice.condition || choice.condition()) {

                        const button = document.createElement('button');

                        button.className = 'choice-btn';

                        button.textContent = choice.text;

                        button.onclick = () => {

                            // 应用选择效果

                            if (choice.effect) {

                                applyEffect(choice.effect);

                            }

                            

                            // 如果是毕业选择

                            if (sceneKey === 'year4-may' && index < 6) {

                                const careers = ['研究生', '外企员工', '教师', '翻译', '创业者', '媒体从业者'];

                                endGame(careers[index])();

                                return;

                            }

                            

                            // 隐藏事件显示

                            document.getElementById('event-display').style.display = 'none';

                            gameState.eventOccurred = false;

                            

                            updateStats();

                        };

                        choicesContainer.appendChild(button);

                    }

                });

            }

            

            // 如果是毕业月，显示毕业选择

            if (sceneKey === 'year4-may') {

                document.getElementById('next-month-btn').style.display = 'none';

            } else {

                document.getElementById('next-month-btn').style.display = 'block';

            }

        }


        // 进入下一个月

        function nextMonth() {

            // 隐藏事件显示

            document.getElementById('event-display').style.display = 'none';

            gameState.eventOccurred = false;

            

            gameState.month++;

            gameState.monthsPassed++;

            

            // 处理学年过渡

            if (gameState.month > 12) {

                gameState.month = 1;

                gameState.year++;

            }

            

            // 特殊月份处理

            if (gameState.month === 9 && gameState.year > 1) {

                // 新学年开始，重置事件标记

                gameState.eventOccurred = false;

            }

            

            // 每过几个月自动增加一些属性

            if (gameState.monthsPassed % 3 === 0) {

                gameState.englishSkill += 2;

                gameState.network += 1;

            }

            

            // 触发随机事件

            showRandomEvent();

            

            updateDisplay();

        }


        // 结束游戏

        function endGame(career) {

            return function() {

                gameState.gameOver = true;

                const careerInfo = careerPaths[career] || careerPaths['default'];

                

                document.getElementById('career-title').textContent = careerInfo.title;

                document.getElementById('career-description').textContent = careerInfo.description;

                document.getElementById('career-result').style.display = 'block';

                document.getElementById('restart-btn').style.display = 'block';

                document.getElementById('choices-container').style.display = 'none';

                document.getElementById('next-month-btn').style.display = 'none';

            };

        }


        // 重新开始游戏

        function restartGame() {

            gameState.year = 1;

            gameState.month = 9;

            gameState.englishSkill = 50;

            gameState.experience = 0;

            gameState.network = 20;

            gameState.finance = 1000;

            gameState.currentScene = 'start';

            gameState.gameOver = false;

            gameState.eventOccurred = false;

            gameState.monthsPassed = 0;

            

            document.getElementById('career-result').style.display = 'none';

            document.getElementById('restart-btn').style.display = 'none';

            document.getElementById('choices-container').style.display = 'grid';

            document.getElementById('next-month-btn').style.display = 'block';

            document.getElementById('event-display').style.display = 'none';

            

            updateDisplay();

        }


        // 初始化游戏

        document.addEventListener('DOMContentLoaded', () => {

            updateDisplay();

            document.getElementById('next-month-btn').addEventListener('click', nextMonth);

            document.getElementById('restart-btn').addEventListener('click', restartGame);

            document.getElementById('restart-btn').style.display = 'none';

        });

    </script>
